## devices.list()

<ReferenceGrid>
  <div>
    <ReferenceHeader title="List devices" method="GET" path="/devices/json" sdk="linxio.devices.list(params)">
      Loads one page of devices.
    </ReferenceHeader>

    <FieldTable
      title="Input"
      fields={[
        {
          name: "params.limit",
          type: "number",
          description: "Number of devices to request.",
        },
        {
          name: "params.page",
          type: "number",
          description: "Page number to request.",
          defaultValue: "1",
        },
        {
          name: "params.fields",
          type: "DeviceField[]",
          description: "Device fields to include in the response projection.",
        },
        {
          name: "params.sort",
          type: "string",
          description: "Sort expression accepted by Linxio.",
        },
      ]}
    />

    <FieldTable
      title="Returns"
      fields={[
        {
          name: "data",
          type: "LinxioDevice[] | null",
          description: "Devices for the requested page.",
        },
        {
          name: "meta",
          type: "LinxioPaginationMeta | null",
          description: "Pagination metadata.",
        },
        {
          name: "error",
          type: "LinxioError | null",
          description: "Typed SDK error when the page fails.",
        },
      ]}
    />
  </div>

  <CodeRail>

```ts title="List devices" lineNumbers
const devices = await linxio.devices.list({
  fields: ["id", "imei", "serial", "vehicle"],
  limit: 100,

if (devices.error) {
  throw devices.error;
}
```

  </CodeRail>
