```ts title="Login and read vehicles" lineNumbers
import { createClient } from "@ambulancewa/linxio-js";

const linxio = createClient();

const login = await linxio.auth.login({
  email: process.env.LINXIO_EMAIL!,
  password: process.env.LINXIO_PASSWORD!,
});

if (login.error) {
  throw login.error;
}

const { data: vehicles, error } = await linxio.vehicles.list({
  fields: ["id", "regNo", "lastLoggedAt"],
  limit: 100,
});

if (error) {
  throw error;
}

console.log(vehicles);
```

## Session Hydration

  <div>
    <ReferenceHeader title="Hydrate a client" sdk="createClient(options)">

const linxio = createClient({
  token: stored.token,
  refreshToken: stored.refreshToken,
});

// Update session tokens at runtime
linxio.setSession({
  token: nextToken,
  refreshToken: nextRefreshToken,
});

// Read the current session

## Automatic Refresh

When a request receives `401` and a refresh token is available, the SDK calls
`/token/refresh`, updates the in-memory session, and retries the original
request once. Concurrent refreshes are coalesced — several simultaneous `401`
responses only trigger one refresh request.

## Two-Factor Authentication

<ReferenceGrid>
  <div>
    <ReferenceHeader
      title="Verify OTP"
      method="POST"
      path="/login/otp"
      sdk="linxio.auth.verifyOtp(payload)"
    >
      Completes an OTP challenge for accounts where Linxio requires a one-time
      code after the initial login. Linxio's integration notes recommend
      disabling OTP for machine users to avoid this step.
    </ReferenceHeader>

    <FieldTable
      title="Method parameters"
      fields={[
        {
          name: "email",
          type: "string",
          required: true,
          description: "Email address for the Linxio user completing the OTP challenge.",
        },
        {
          name: "code",
          type: "string",
          required: true,
          description: "One-time code delivered by Linxio to the user's registered device.",
        },
      ]}
    />
  </div>

  <CodeRail>

```ts title="Verify an OTP challenge" lineNumbers
const login = await linxio.auth.login({
  email: process.env.LINXIO_EMAIL!,
  password: process.env.LINXIO_PASSWORD!,
});

if (login.data?.otpRequired) {
  const otp = await linxio.auth.verifyOtp({
    email: process.env.LINXIO_EMAIL!,
    code, // delivered out of band
  });

  if (otp.error) {
    throw otp.error;
  }
}
```

</ReferenceGrid>

## Secrets

Do not commit credentials. Prefer environment variables, a secret manager, or
your deployment platform's encrypted environment configuration.

```ts title="Recommended credential pattern"
const linxio = createClient();

await linxio.auth.login({
  email: process.env.LINXIO_EMAIL!,    // from env, not source
  password: process.env.LINXIO_PASSWORD!, // from env, not source
});
```
